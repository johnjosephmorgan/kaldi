#!/usr/bin/env bash
datadir=~/mflts
wavlist=$(find $datadir -type f -name "*sif.wav" | shuf -n 2)
{
  while read line; do
    bn=$(basename "$line" .wav)
    mkdir -p flacs
    sox "$line" -c 1 -r 16000 flacs/$bn.flac
done
